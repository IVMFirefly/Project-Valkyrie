####Stability, Laws and Popularity events - Levi########

add_namespace = politics

country_event = {#loss of stability

	id = politics.1 
	title = politics.1.t 
	desc = politics.1.d 
	
	trigger = {
	
		OR = {
		
			AND = {
			
				has_idea = very_high_stability
				
				has_political_power < 100
			
			}
			
			AND = {
			
				has_idea = high_stability
				
				has_political_power < 0
			
			}
			
			AND = {
			
				has_idea = normal_stability
				
				has_political_power < -100
			
			}
			
			AND = {
			
				has_idea = low_stability
				
				has_political_power < -200
			
			}
		
		}
	
	
	}
	
	mean_time_to_happen = { days = 70 }
	
	option = {
	
		name = politics.1.a 
		
		stability_downgrade = yes 
		
		custom_effect_tooltip = stability_downgrade_tooltip
	
	}

}

country_event = {#gain of stability

	id = politics.2
	title = politics.2.t 
	desc = politics.2.d 
	
	trigger = {
	
		OR = {
		
			AND = {
			
				has_idea = very_low_stability
				
				has_political_power > 0
			
			}
			
			AND = {
			
				has_idea = low_stability
				
				has_political_power > 100
			
			}
			
			AND = {
			
				has_idea = normal_stability
				
				has_political_power > 200
			
			}
			
			AND = {
			
				has_idea = high_stability
				
				has_political_power > 300
			
			}
		
		}
	
	
	}
	
	mean_time_to_happen = { days = 70 }
	
	option = {#take it 
	
		name = politics.2.a 
		
		ai_chance = { factor = 80 }
		
		stability_upgrade = yes 
		
		custom_effect_tooltip = stability_upgrade_tooltip
		
		add_political_power = -150 
	
	}
	
	option = {#don't take it
	
		name = politics.2.b 
		
		ai_chance = { factor = 20 }
		
		add_political_power = 50 
	
	}

}

country_event = {#Popularity change event 

	id = politics.3
	title = politics.3.t 
	desc = politics.3.d 
	
	trigger = {
	
		OR = {
		
			AND = {
			
				very_low_popularity_trigger = yes 
				
				NOT = { has_idea = very_low_popularity }
				
			}
			
			AND = {
			
				low_popularity_trigger = yes 
				
				NOT = { has_idea = low_popularity }
				
			}
			
			AND = {
			
				normal_popularity_trigger = yes 
				
				NOT = { has_idea = normal_popularity }
				
			}
			
			AND = {
			
				high_popularity_trigger = yes 
				
				NOT = { has_idea = high_popularity }
				
			}
			
			AND = {
			
				very_high_popularity_trigger = yes 
				
				NOT = { has_idea = very_high_popularity }
				
			}
		
		}
		
	}
	
	option = {
	
		name = politics.3.a 
		
				
		if = {
		
			limit = {
			
				very_low_popularity_trigger = yes 
			
			}
			
			remove_ideas_with_trait = politics_trait
			
			add_ideas = very_low_popularity
		
		}

		
		if = {
		
			limit = {
			
				low_popularity_trigger = yes 
			
			}
			
			remove_ideas_with_trait = politics_trait
			
			add_ideas = low_popularity
		
		}
		
		if = {
		
			limit = {
			
				normal_popularity_trigger = yes 
			
			}
			
			remove_ideas_with_trait = politics_trait
			
			add_ideas = normal_popularity
		
		}
		
		if = {
		
			limit = {
			
				high_popularity_trigger = yes 
			
			}
			
			remove_ideas_with_trait = politics_trait
			
			add_ideas = high_popularity
		
		}
		
		if = {
		
			limit = {
			
				very_high_popularity_trigger = yes 
			
			}
			
			remove_ideas_with_trait = politics_trait
			
			add_ideas = very_high_popularity
		
		}
	
	}

}
